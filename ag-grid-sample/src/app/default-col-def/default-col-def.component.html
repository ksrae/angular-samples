DefailtColDef: 모든 column에 적용되는 기본 값

  
    autoHeaderHeight: boolean, // header 높이 조절
    autoHeight: boolean, // column 높이 조절
    cellEditorPopup: boolean, // edit 할 때 살짝 떠오른 input이 표시됨
    filter: boolean, // filter 기능을 추가. header 우측에 햄버거 메뉴가 표시됨. custom 가능
    minWidth: number, // 기본 길이, flex를 설정하면 의미 없음
    resizable: boolean, // column resize 지원 여부
    sortable: boolean, // header에 화살표가 표시됨
    editable: boolean, // double click 하면 edit 할 수 있음
    suppressMenu: boolean, // true: header의 햄버거 메뉴 제거, default는 false
    lockVisible: true, // 알 수 없음
    lockPinned: true, // 알 수 없음
    lockPosition: true // true: drag 안됨, default는 false
